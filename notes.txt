package main

import (
	"fmt"
	"log"
	"net/smtp"
)

func main() {
	// Define the SMTP server
	smtpHost := "mail.your-server.de" // Hetzner's outgoing mail server
	smtpPort := "587"                 // Standard SMTP port (for STARTTLS)

	// Define the sender and recipient email
	from := "denis.kantic@vortexdigitalsystems.com" // Your Hetzner email address
	to := []string{"denis.kantic18@gmail.com"}      // Recipient's email address

	// Define the email message
	subject := "Subject: Test Email from Golang"
	body := "This is a test email sent from Golang using Hetzner SMTP!"

	// Construct the email message
	message := []byte(fmt.Sprintf("To: %s\r\nFrom: %s\r\nSubject: %s\r\n\r\n%s", to[0], from, subject, body))

	// SMTP authentication
	auth := smtp.PlainAuth("", from, "Nightshade1", smtpHost)

	// Send the email
	err := smtp.SendMail(smtpHost+":"+smtpPort, auth, from, to, message)
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println("Email sent successfully!")
}
